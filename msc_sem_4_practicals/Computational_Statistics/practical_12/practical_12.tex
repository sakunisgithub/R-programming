\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{Question}}

\hspace{1cm} Consider the data-set representing lifetimes (in hours) of a system component $$X = \{12, 15, 18, 21, 24, 30, 35, 40\}$$ and let $\theta = \dfrac{\mu}{\sigma}$ where $\mu$ and $\sigma$ denote the population mean and population standard deviation respectively.

\begin{enumerate}[(i)]
\item Treating this sample as an empirical distribution, apply the jackknife method to compute estimates $\widehat{\theta}_{(i)} = \dfrac{\bar{X}_{(i)}}{S_{(i)}}$.

\item Using these leave-one-out estimates, provide jackknife estimate of bias and standard error of $\widehat{\theta}$.

\item Construct a $95\%$ confidence interval for $\theta$.

\item Provide a visualization of the jackknife estimates.
\end{enumerate}

\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{Theory}}

\begin{itemize}
\item Remove $i$-th observation $X_i$ and compute the estimate based on remaining $n-1$ observations as $$\widehat{\theta}_{(i)} = \dfrac{\bar{X}_{(i)}}{S_{(i)}}.$$

\item Repeat the above for $i = 1(1)n$ to get $n$ jackknife estimates $$\widehat{\theta}_{(1)}, \widehat{\theta}_{(2)}, \ldots, \widehat{\theta}_{(n)}.$$
\end{itemize}

$\bullet$ \textbf{Jackknife estimate of Bias \& standard error} of $\hat{\theta}$ are given by $$\text{Bias}_{\text{jack}} = (n - 1) \cdot \left( \dfrac{1}{n} \sum \limits_{i = 1}^{n} \widehat{\theta}_{(i)} - \hat{\theta} \right) \qquad \& \qquad \text{se}_{\text{jack}} = \sqrt{\dfrac{n-1}{n} \sum \limits_{i = 1}^{n} \left[\widehat{\theta}_{(i)} - \dfrac{1}{n} \sum \limits_{i = 1}^{n} \widehat{\theta}_{(i)}\right]^2}$$

\begin{align*}
\text{where } \hat{\theta} &: \text{ estimate of } \theta \text{ from the original sample}, \\[0.2em]
\widehat{\theta}_{(i)} &: \text{ estimate of } \theta \text{ from } i\text{-th jackknife sample}, \\[0.2em]
n &: \text{size of original sample}.
\end{align*}

\newpage

$\bullet$ \textbf{Bias-reduced Jackknife estimate} of $\hat{\theta}$ is given by 
\begin{align*}
\widehat{\theta}_{\text{jack}} = \widehat{\theta} - \text{Bias}_{\text{jack}} &= \widehat{\theta} -  (n - 1) \cdot \left( \dfrac{1}{n} \sum \limits_{i = 1}^{n} \widehat{\theta}_{(i)} - \hat{\theta} \right) \\[0.5em]
&= n \widehat{\theta} - (n - 1) \cdot \dfrac{1}{n} \sum \limits_{i = 1}^{n} \widehat{\theta}_{(i)} \\[1em]
&= n \widehat{\theta} - (n - 1) \widehat{\theta}_{(\cdot)}
\end{align*}

\begin{align*}
\text{where } \widehat{\theta}_{(\cdot)} &: \text{average of } n \text{ jackknife estimates.}
\end{align*}

$\bullet$ Assuming approximate normality, $100(1 - \alpha)\%$ confidence interval for $\theta$ is given by

$$\left( \widehat{\theta}_{\text{jack}} - \tau_{\alpha / 2} \cdot \text{se}_{\text{jack}}, \widehat{\theta}_{\text{jack}} + \tau_{\alpha / 2} \cdot \text{se}_{\text{jack}} \right).$$
