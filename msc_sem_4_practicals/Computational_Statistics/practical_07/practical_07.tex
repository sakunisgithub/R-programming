\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{Question 1}}

\hspace{1cm} A continuous random variable $X$ has the following probability density function :

\begin{align*}
f(x) &= \begin{cases}
            3x^2, \,\, 0 < x < 1 \\[0.2em]
            0, \,\, \text{otherwise}
        \end{cases}
\end{align*}
\begin{enumerate}[(i)]
\item Verify that $f(\cdot)$ is a valid probability distribution. Derive the Cumulative Distribution Function $F(\cdot)$. Obtain the inverse transformation $F^{-1}(\cdot)$.

\item Generate $10000$ random observations using the inverse transform method. Plot histogram (probability scale) and overlay theoretical density curve. Compute empirical mean and variance, theoretical mean and variance and compare the results.
\end{enumerate}

\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{Theory}}

\begin{itemize}
\item[\scalebox{2.0}{\twemoji{keycap: 1}}] \hspace{0.1cm} To verify $f(\cdot)$ as a valid PDF, we integrate 

\begin{align*}
\int \limits_{-\infty}^{\infty} f(x) dx &= \int \limits_{0}^{1} 3x^2 dx \\[0.2em]
&= 3 \left[ \dfrac{x^3}{3} \right]_{0}^{1} \\[0.2em]
&= 1.
\end{align*}

Clearly $f(\cdot)$ is non-negative. And it integrates to unity in $\mathbb{R}$. Hence $f(\cdot)$ is indeed a valid PDF. \\[0.3em]

The corresponding Cumulative Distribution Function can be obtained as

\begin{align*}
F(x) &= \int \limits_{-\infty}^{x} f(x) dx \\[0.3em]
&= \int \limits_{0}^{x} 3t^2 dt, \,\, 0 < x < 1 \\[0.3em]
&= x^3, \,\, 0 < x < 1.
\end{align*}

Thus,
\begin{align*}
F(x) &= \begin{cases}
          0, \,\, x < 0 \\[0.2em]
          x^3, \,\, 0 \leq x < 1 \\[0.2em]
          1, \,\, x \geq 1.
        \end{cases}
\end{align*}

To obtain $F^{-1}$, we solve

\begin{align*}
x^3 &= y \\[0.2em]
\Rightarrow x &= y^{\frac{1}{3}}
\end{align*}

Thus, $F^{-1}(x) = x^{\frac{1}{3}}$, $0 < x < 1$.

\item[\scalebox{2.0}{\twemoji{keycap: 2}}] \hspace{0.1cm} For inverse transform method, we 

\begin{enumerate}[I.]
\item Generate random numbers $u_1, u_2, \ldots, u_n \sim U(0, 1)$.
\item Evaluate $x_i = F^{-1}(u_i) \,\, \forall i = 1(1)n$.
\end{enumerate}
\end{itemize}

\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{Question 2}}

\hspace{1cm} Generate $5000$ observations from $Exp(\lambda = 2)$ by inverse transform method. Plot the histogram and the theoretical density. Compute and compare theoretical and empirical mean and variance.

\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{Theory}}

For inverse transform method, we 

\begin{enumerate}[I.]
\item Generate random numbers $u_1, u_2, \ldots, u_n \sim U(0, 1)$.
\item Evaluate $x_i = - \dfrac{1}{2} \log(1 - u_i) \,\, \forall i = 1(1)n$ or $x_i = \dfrac{1}{2} \log(1 - u_i)$ equivalently.
\end{enumerate}
