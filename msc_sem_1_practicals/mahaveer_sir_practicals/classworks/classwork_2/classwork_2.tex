\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{Question}}

\hspace{1cm} For $n = \{5, 10, 15, 20, 25, 30\}$ calculate the value of $$\sqrt{1 + 2 \sqrt{1 + 3 \sqrt{1 + 4 \sqrt{1 + 5 \sqrt{1 + \ldots \sqrt{1 + (n-1)\sqrt{1 + n}}}}}}}$$ .





\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{R Program}}

\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hldef{nested_square_root} \hlkwb{<-} \hlkwa{function}\hldef{(}\hlkwc{n}\hldef{)\{}

  \hldef{result} \hlkwb{<-} \hlkwd{sqrt}\hldef{(}\hlnum{1} \hlopt{+} \hldef{n)}

  \hlkwa{for}\hldef{(i} \hlkwa{in} \hldef{(n}\hlopt{-}\hlnum{1}\hldef{)}\hlopt{:}\hlnum{2}\hldef{)\{}
    \hldef{result} \hlkwb{<-} \hlkwd{sqrt}\hldef{(}\hlnum{1} \hlopt{+} \hldef{i} \hlopt{*} \hldef{result)}
  \hldef{\}}

  \hlkwd{return}\hldef{(result)}
\hldef{\}}
\end{alltt}
\end{kframe}
\end{knitrout}
