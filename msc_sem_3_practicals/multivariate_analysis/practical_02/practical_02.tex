\scalebox{2.0}{\twemoji{bullseye}} \textcolor{blue}{\textbf{Question}}

\begin{enumerate}
\item Determine the correlation matrix $R = ((\rho_{ij}))$, if $N = 25$ and
\begin{gather*}
S = 
\begin{bmatrix}
95.2933 & 52.8683 & 69.6617 & 46.1117 \\
52.8683 & 54.3660 & 51.3117 & 35.0533 \\
69.6617 & 51.3117 & 100.8067 & 56.5400 \\
46.1117 & 35.0533 & 56.5400 & 45.0233
\end{bmatrix}.
\end{gather*}

\item Calculate sample mean vector $\utilde{\bar{X}}$ and sample variance-covariance matrix $S$ for the following data.
\begin{table}[!htbp]
\def\arraystretch{1.5}

\begin{center}
\begin{tabular}{|>{\centering}m{2cm}|>{\centering}m{2cm}|>{\centering\arraybackslash}m{2cm}|}

\hline
Patient Number & Drug $A$ & Drug $B$ \\
\hline
\hline
$1$ & $1.9$ & $0.7$ \\
\hline
$2$ & $0.8$ & $-1.6$ \\
\hline
$3$ & $1.1$ & $-0.2$ \\
\hline
$4$ & $0.1$ & $-0.2$ \\
\hline
$5$ & $-0.1$ & $-0.1$ \\
\hline
$6$ & $4.4$ & $3.4$ \\
\hline
$7$ & $5.5$ & $3.7$ \\
\hline
$8$ & $1.6$ & $0.8$ \\
\hline
$9$ & $4.6$ & $0$ \\
\hline
$10$ & $3.9$ & $2.0$ \\
\hline
\end{tabular}
\end{center}
\end{table}

\end{enumerate}

\scalebox{2.0}{\twemoji{keycap: 1}} \hspace{0.1cm} Calculating correlation matrix : $R = \Lambda \cdot S \cdot \Lambda$ where $\Lambda$ is a diagonal matrix with elements being $\dfrac{1}{\sigma_{ii}}$.

\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hldef{S} \hlkwb{<-} \hlkwd{matrix}\hldef{(}\hlkwd{c}\hldef{(}\hlnum{95.2933}\hldef{,} \hlnum{52.8683}\hldef{,} \hlnum{69.6617}\hldef{,} \hlnum{46.1117}\hldef{,}
             \hlnum{52.8683}\hldef{,} \hlnum{54.3660}\hldef{,} \hlnum{51.3117}\hldef{,} \hlnum{35.0533}\hldef{,}
             \hlnum{69.6617}\hldef{,} \hlnum{51.3117}\hldef{,} \hlnum{100.8067}\hldef{,} \hlnum{56.5400}\hldef{,}
             \hlnum{46.1117}\hldef{,} \hlnum{35.0533}\hldef{,} \hlnum{56.5400}\hldef{,} \hlnum{45.0233}\hldef{),} \hlkwc{nrow} \hldef{=} \hlnum{4}\hldef{,} \hlkwc{ncol} \hldef{=} \hlnum{4}\hldef{,} \hlkwc{byrow} \hldef{=} \hlnum{TRUE}\hldef{)}
\end{alltt}
\end{kframe}
\end{knitrout}
